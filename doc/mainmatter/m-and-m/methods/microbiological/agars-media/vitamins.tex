\subsubsection{Vitamins Solution (RPMI 1640)}

\begin{itemize}
	\item \SIg{0.02} \textsc{d}-Biotin
	\item \SIg{0.3} Choline chloride
	\item \SIg{0.1} Folic acid
	\item \SIg{3.5} myo-Inositol
	\item \SIg{0.1} Niacinamide
	\item \SIg{0.1} p-Aminobenzoic acid
	\item \SIg{0.025} \textsc{d}-Pantothenic \ce{acid*0.5Ca}
	\item \SIg{0.1} \ce{Pyridoxal*HCl}
	\item \SIg{0.02} Riboflavin
	\item \SIg{0.1} \ce{Thiamine*HCl}
	\item \SIg{0.0005} Vitamin B12
	\item \SIg{0.2} \ce{KCl}
	\item \SIg{0.2} \ce{KH2PO4}
	\item \SIg{8.0} \ce{NaCl}

\end{itemize}
